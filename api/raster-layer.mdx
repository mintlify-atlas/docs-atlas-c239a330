---
title: RasterLayer
description: Renders raster tiles on the map
---

RasterLayer displays raster imagery from a raster tile source.

[Mapbox Style Specification](https://docs.mapbox.com/mapbox-gl-js/style-spec/layers/#raster)

## Props

<ParamField path="id" type="string" required>
  A unique identifier for the layer.
</ParamField>

<ParamField path="existing" type="boolean">
  If true, references an existing layer in the style.
</ParamField>

<ParamField path="sourceID" type="string" default="Mapbox.StyleSource.DefaultSourceID">
  The source from which to obtain data.
</ParamField>

<ParamField path="sourceLayerID" type="string">
  Identifier of the layer within the source.
</ParamField>

<ParamField path="aboveLayerID" type="string">
  Inserts the layer above the specified layer ID.
</ParamField>

<ParamField path="belowLayerID" type="string">
  Inserts the layer below the specified layer ID.
</ParamField>

<ParamField path="slot" type="'bottom' | 'middle' | 'top'">
  The slot this layer is assigned to. v11 only
</ParamField>

<ParamField path="minZoomLevel" type="number">
  Minimum zoom level for layer visibility.
</ParamField>

<ParamField path="maxZoomLevel" type="number">
  Maximum zoom level for layer visibility.
</ParamField>

<ParamField path="filter" type="FilterExpression">
  Filters features based on a condition.
</ParamField>

<ParamField path="style" type="RasterLayerStyleProps" required>
  Customizable style attributes.
</ParamField>

## Style Properties

<ParamField path="style.visibility" type="'visible' | 'none'" default="visible">
  Controls layer visibility.
</ParamField>

<ParamField path="style.rasterOpacity" type="number" default="1">
  The opacity at which the image will be drawn. Range: 0-1.
</ParamField>

<ParamField path="style.rasterHueRotate" type="number" default="0">
  Rotates hues around the color wheel in degrees.
</ParamField>

<ParamField path="style.rasterBrightnessMin" type="number" default="0">
  Increase or reduce the brightness of the image. Range: 0-1.
</ParamField>

<ParamField path="style.rasterBrightnessMax" type="number" default="1">
  Increase or reduce the brightness of the image. Range: 0-1.
</ParamField>

<ParamField path="style.rasterSaturation" type="number" default="0">
  Increase or reduce the saturation of the image. Range: -1 to 1.
</ParamField>

<ParamField path="style.rasterContrast" type="number" default="0">
  Increase or reduce the contrast of the image. Range: -1 to 1.
</ParamField>

<ParamField path="style.rasterResampling" type="'linear' | 'nearest'" default="linear">
  The resampling/interpolation method to use for overscaling.
</ParamField>

## Example

```tsx
import { MapView, RasterSource, RasterLayer } from '@rnmapbox/maps';

function Map() {
  return (
    <MapView>
      <RasterSource
        id="satellite"
        tileUrlTemplates={['https://tiles.example.com/{z}/{x}/{y}.png']}
        tileSize={256}
      >
        <RasterLayer
          id="satellite-layer"
          style={{
            rasterOpacity: 0.8,
            rasterSaturation: 0.2,
          }}
        />
      </RasterSource>
    </MapView>
  );
}
```
