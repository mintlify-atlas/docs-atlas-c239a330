# Image

The Image component allows you to register a single image with the map using a React Native View as the source. This is useful for creating custom marker icons from React components.

## Import

```tsx
import { Image } from '@rnmapbox/maps';
```

## Basic Usage

```tsx
import { MapView, Images, Image } from '@rnmapbox/maps';
import { View, Text } from 'react-native';

function MapWithCustomImage() {
  return (
    <MapView style={{ flex: 1 }}>
      <Images>
        <Image name="customMarker">
          <View style={{ width: 30, height: 30, backgroundColor: 'red' }}>
            <Text>📍</Text>
          </View>
        </Image>
      </Images>
    </MapView>
  );
}
```

## Props

### name

<ParamField path="name" type="string" required>
  ID of the image. This name is used to reference the image in layer styles.

  ```tsx
  <Image name="myMarker">
    <View style={{ width: 30, height: 30 }} />
  </Image>
  ```
</ParamField>

### sdf

<ParamField path="sdf" type="boolean">
  Make image an SDF (Signed Distance Field) icon. SDF icons can be recolored using style properties.
  
  See [SDF icons documentation](https://docs.mapbox.com/help/troubleshooting/using-recolorable-images-in-mapbox-maps/) for more information.

  ```tsx
  <Image name="sdfIcon" sdf={true}>
    <View style={{ width: 30, height: 30 }} />
  </Image>
  ```
</ParamField>

### stretchX

<ParamField path="stretchX" type="[number, number][]">
  An array of two-element arrays, consisting of two numbers that represent the from position and the to position of areas that can be stretched horizontally.
  
  Useful for creating resizable UI elements like buttons or speech bubbles.

  ```tsx
  <Image name="button" stretchX={[[10, 20], [30, 40]]}>
    <View style={{ width: 50, height: 30 }} />
  </Image>
  ```
</ParamField>

### stretchY

<ParamField path="stretchY" type="[number, number][]">
  An array of two-element arrays, consisting of two numbers that represent the from position and the to position of areas that can be stretched vertically.
  
  Useful for creating resizable UI elements like buttons or speech bubbles.

  ```tsx
  <Image name="button" stretchY={[[10, 20]]}>
    <View style={{ width: 50, height: 30 }} />
  </Image>
  ```
</ParamField>

### content

<ParamField path="content" type="[number, number, number, number]">
  An array of four numbers, with the first two specifying the left-top corner, and the last two specifying the right-bottom corner.
  
  If present, and if the icon uses `icon-text-fit`, the symbol's text will be fit inside the content box.

  ```tsx
  <Image name="textBox" content={[5, 5, 45, 25]}>
    <View style={{ width: 50, height: 30 }} />
  </Image>
  ```
</ParamField>

### scale

<ParamField path="scale" type="number">
  Scale factor for the image. Use values greater than 1 for higher resolution displays.

  ```tsx
  <Image name="hdIcon" scale={2}>
    <View style={{ width: 60, height: 60 }} />
  </Image>
  ```
</ParamField>

### children

<ParamField path="children" type="ReactElement" required>
  Single React Native view rendering the image. This can be any React Native component.

  ```tsx
  <Image name="marker">
    <View style={{ width: 30, height: 30, backgroundColor: 'blue' }} />
  </Image>
  ```
</ParamField>

## Methods

### refresh()

```tsx
refresh(): void
```

Forces the image to re-render. Useful when the child view has changed and you need to update the image on the map.

```tsx
import { useRef } from 'react';
import { Image } from '@rnmapbox/maps';

function MyComponent() {
  const imageRef = useRef<Image>(null);
  
  const updateImage = () => {
    // Update the view content somehow
    // Then refresh the image
    imageRef.current?.refresh();
  };
  
  return (
    <Image name="marker" ref={imageRef}>
      <DynamicView />
    </Image>
  );
}
```
